CREATE GPIO tInput
SET tInput MODE INPUT
SET tInput PIN 5
SET tInput VAR tInputVar
PRINT tInput

CREATE GPIO tOutput
SET tOutput MODE OUTPUT
SET tOutput PIN 6
SET tOutput VAR tOutputVar
PRINT tOutput

CREATE TIMER tPump
SET tPump MODE ONDelay
SET tPump TIME 2
SET tPump INPUT tInput
SET tPump OUTPUT tOutput
SET tPump ENABLE
PRINT tPump

SET comm ID 0
SET comm OUT iWaterLevel WaterLevel
